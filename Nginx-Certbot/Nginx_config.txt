# Default server configuration
#
server {

    server_name www.ronaldlepape.fr;

    # Main app reverse proxy
    location / {
        proxy_pass http://127.0.0.1:31234;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }

    # Redirect to the starting page
    location = /prometheus {
    return 302 /prometheus/query;}

    # Prometheus reverse proxy
    location /prometheus/ {
        proxy_pass http://127.0.0.1:30900/;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header Accept-Encoding "";

        # Needed to rewrite links so that Prometheus still works under subpath
        sub_filter_once off;
        sub_filter 'href="/' 'href="/prometheus/';
        sub_filter 'src="/' 'src="/prometheus/';
    }

    # Grafana reverse proxy
    location /grafana/ {
        proxy_pass http://127.0.0.1:32000/;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header Accept-Encoding "";

        # Needed to rewrite links so that Grafana works under a subpath
        # sub_filter_once off;
        # sub_filter 'href="/' 'href="/grafana/';
        # sub_filter 'src="/' 'src="/grafana/';
    }

    listen [::]:443 ssl ipv6only=on; # managed by Certbot
    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/www.ronaldlepape.fr/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/www.ronaldlepape.fr/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot
}
